Why this lab exists

Module 5's three theory topics gave you the mental model: when to pick pgvector over Mosaic AI Vector Search (5.1), how to design the schema and pick HNSW parameters (5.2), and why hybrid retrieval with Reciprocal Rank Fusion beats pure vector (5.3). This lab makes each of those concrete. You will reconnect to the Module 3 project, install pgvector, build the documents schema, embed a real corpus, prove that hybrid wins on a deliberate corner case, and wrap it all in a rag() function.

Total runtime ~75 minutes, of which about 8 minutes is your hands on the keyboard and the rest is embedding API calls and index builds. Use the wait times to skim Module 5 theory if you haven't.

▸ Run mode

Do this lab in a Databricks notebook attached to a serverless cluster (or DBR 14+). All cells are Python except where SQL is explicitly called out via %sql. Copy each cell into your own notebook as you go — every code block on this page has a Copy button.

STEP 02

Build the documents schema

The schema from Module 5 theory 5.2: a vector(1024) column for embeddings, a generated tsvector for BM25, JSONB metadata for filter pushdown, and three indexes (HNSW + 2× GIN). All three indexes are required for the hybrid retrieval planner to be happy.

2aCreate the documents table
Cell 4 · schema DDLPython 
# Cell 4 — schema DDL
DDL = """
CREATE TABLE IF NOT EXISTS documents (
    id          BIGSERIAL PRIMARY KEY,
    source      TEXT,
    title       TEXT,
    content     TEXT NOT NULL,
    metadata    JSONB DEFAULT '{}'::jsonb,
    embedding   vector(1024),
    tsv         tsvector GENERATED ALWAYS AS
                (to_tsvector('english',
                    coalesce(title,'') || ' ' || coalesce(content,''))) STORED,
    created_at  TIMESTAMPTZ DEFAULT now()
);
"""

with engine.begin() as conn:
    conn.execute(text(DDL))

with engine.connect() as conn:
    cols = conn.execute(text("""
        SELECT column_name, data_type
        FROM information_schema.columns
        WHERE table_name = 'documents'
        ORDER BY ordinal_position
    """)).fetchall()
    print("✅ documents table created:")
    for c in cols:
        print(f"   · {c.column_name:<12} {c.data_type}")

Cell 7 · embed via Foundation Models + bulk-loadPython 
# Cell 7 — embed via Foundation Models API and bulk-load
client = w.serving_endpoints.get_open_ai_client()

def embed(texts, batch=8):
    """Embed a list of strings; returns a list of vectors (1024-dim)."""
    out = []
    for i in range(0, len(texts), batch):
        chunk = texts[i:i+batch]
        r = client.embeddings.create(
            model=TUTORIAL["embed_model"],
            input=chunk,
        )
        out.extend(d.embedding for d in r.data)
    return out

# Build the input strings: title + content gives the model both signals
inputs = [f"{title}\n{content}" for (_, title, content, _) in DOCS]
print(f"Embedding {len(inputs)} docs in batches of 8...")
vectors = embed(inputs)
print(f"✅ Got {len(vectors)} embeddings, each {len(vectors[0])}-dim")

# Bulk-load in a single transaction
INSERT_SQL = """
INSERT INTO documents (source, title, content, metadata, embedding)
VALUES (:source, :title, :content, :metadata, :embedding)
ON CONFLICT DO NOTHING
"""

with engine.begin() as conn:
    for (source, title, content, meta), vec in zip(DOCS, vectors):
        conn.execute(text(INSERT_SQL), {
            "source":    source,
            "title":     title,
            "content":   content,
            "metadata":  json.dumps(meta),
            "embedding": str(vec),  # pgvector accepts a string repr of a list
        })

with engine.connect() as conn:
    n = conn.execute(text("SELECT count(*) FROM documents")).scalar()
    print(f"\n✅ {n} docs in the table")

STEP 04

Pure vector retrieval (the baseline)

Establish what pure vector search is good at — and where it breaks. The first query is a semantic question (vector wins). The second is a SKU lookup (vector misses).

Cell 8 · pure vector retrieval helperPython 
# Cell 8 — pure vector retrieval
def vector_search(question, k=5):
    qv = embed([question])[0]
    with engine.connect() as conn:
        rows = conn.execute(text("""
            SELECT id, title,
                   embedding <=> CAST(:qv AS vector) AS distance
            FROM documents
            ORDER BY embedding <=> CAST(:qv AS vector)
            LIMIT :k
        """), {"qv": str(qv), "k": k}).fetchall()
    return rows

print("Q: How do I return a broken laptop?\n")
for r in vector_search("How do I return a broken laptop?", k=5):
    print(f"   [{r.distance:.4f}]  {r.title}")

STEP 05

Hybrid retrieval with Reciprocal Rank Fusion

Run vector and BM25 in parallel, then fuse with RRF. Score formula: 1/(60+rank) summed across both legs. The query plan uses HNSW for the v CTE and GIN-on-tsv for the k CTE — both indexes from Step 2 earn their keep.

Cell 10 · hybrid retrieval with RRFPython 
# Cell 10 — hybrid retrieval with RRF
def hybrid_search(question, k=5, fanout=50):
    qv = embed([question])[0]
    SQL = """
    WITH v AS (
      SELECT id, ROW_NUMBER() OVER (ORDER BY embedding <=> CAST(:qv AS vector)) AS rk
      FROM documents
      ORDER BY embedding <=> CAST(:qv AS vector)
      LIMIT :fanout
    ),
    k AS (
      SELECT id, ROW_NUMBER() OVER
        (ORDER BY ts_rank(tsv, plainto_tsquery('english', :q)) DESC) AS rk
      FROM documents
      WHERE tsv @@ plainto_tsquery('english', :q)
      LIMIT :fanout
    )
    SELECT d.id, d.title,
           COALESCE(1.0/(60+v.rk), 0) AS v_score,
           COALESCE(1.0/(60+k.rk), 0) AS k_score,
           COALESCE(1.0/(60+v.rk), 0) + COALESCE(1.0/(60+k.rk), 0) AS rrf
    FROM documents d
    LEFT JOIN v USING(id)
    LEFT JOIN k USING(id)
    WHERE v.id IS NOT NULL OR k.id IS NOT NULL
    ORDER BY rrf DESC
    LIMIT :k
    """
    with engine.connect() as conn:
        rows = conn.execute(text(SQL), {
            "qv": str(qv), "q": question, "k": k, "fanout": fanout
        }).fetchall()
    return rows

print("Q: What's the warranty on the LB-9047X?  (HYBRID)\n")
print(f"   {'TITLE':<40} {'VEC':>8} {'BM25':>8} {'RRF':>8}")
print(f"   {'-'*40} {'-'*8} {'-'*8} {'-'*8}")
for r in hybrid_search("What's the warranty on the LB-9047X?", k=5):
    print(f"   {r.title[:40]:<40} {r.v_score:>8.4f} {r.k_score:>8.4f} {r.rrf:>8.4f}")

STEP 06

Wrap it in a rag() function

Embed the question → run hybrid retrieval → build a context block from the top-k → call the chat model with a system prompt that pins it to the retrieved evidence. ~40 lines, no framework.

Cell 12 · end-to-end RAG functionPython 
# Cell 12 — end-to-end RAG function
def rag(question: str, k: int = 4) -> dict:
    """Hybrid-retrieval RAG. Returns dict with answer + sources."""
    # 1. Retrieve top-k via hybrid
    rows = hybrid_search(question, k=k)
    if not rows:
        return {"answer": "No relevant documents found.", "sources": []}

    # 2. Build context block
    with engine.connect() as conn:
        docs = conn.execute(text("""
            SELECT id, title, content, source
            FROM documents WHERE id = ANY(:ids)
        """), {"ids": [r.id for r in rows]}).fetchall()
        order = {r.id: i for i, r in enumerate(rows)}
        docs = sorted(docs, key=lambda d: order[d.id])

    context = "\n\n".join(
        f"# {d.title}\n[source: {d.source}]\n{d.content}" for d in docs
    )

    # 3. Call the chat model
    out = client.chat.completions.create(
        model=TUTORIAL["chat_model"],
        messages=[
            {"role": "system", "content":
                "Answer the user's question using ONLY the provided context. "
                "If the answer isn't in the context, say so. Cite source paths "
                "in brackets when you use a fact."},
            {"role": "user", "content":
                f"Context:\n{context}\n\nQuestion: {question}"}
        ],
        temperature=0.1,
    )
    return {
        "answer":  out.choices[0].message.content,
        "sources": [{"title": d.title, "source": d.source} for d in docs],
    }
